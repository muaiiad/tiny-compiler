/* Sample program in Tiny language –
   computes factorial */

int main()
{
    int x;

    read x;  /* input an integer */

    if x > 0 then       /* don’t compute if x <= 0 */
        int fact := 1;

        repeat
            fact := fact * x;
            x := x - 1;
        until x = 0

        write fact;    /* output factorial of x */
    end

    return 0;
}
